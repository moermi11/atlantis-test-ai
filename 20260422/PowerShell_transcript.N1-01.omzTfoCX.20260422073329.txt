#!/bin/sh
set -eu

SCRIPT_DIR=$(CDPATH= cd -- "$(dirname -- "$0")" && pwd)
REPO_ROOT=$(CDPATH= cd -- "$SCRIPT_DIR/../.." && pwd)
PACKAGES_GLOB_DIR=${PACKAGES_GLOB_DIR:-"$REPO_ROOT/packages"}
REPO_DIR=${REPO_DIR:-"$REPO_ROOT/out/repo"}

if ! command -v dpkg-scanpackages >/dev/null 2>&1; then
    echo "dpkg-scanpackages is required (package: dpkg-dev)." >&2
    exit 1
fi

mkdir -p "$REPO_DIR"
find "$REPO_DIR" -maxdepth 1 -type f \( -name '*.deb' -o -name 'Packages' -o -name 'Packages.gz' \) -delete

found_debs=0
for deb in "$PACKAGES_GLOB_DIR"/*.deb; do
    if [ -f "$deb" ]; then
        cp "$deb" "$REPO_DIR"/
        found_debs=1
    fi
done

if [ "$found_debs" -ne 1 ]; then
    echo "No built Atlantis .deb packages found under $PACKAGES_GLOB_DIR." >&2
    echo "Build packages first with dpkg-buildpackage before generating the feed." >&2
    exit 1
fi

(
    cd "$REPO_DIR"
    dpkg-scanpackages --multiversion . /dev/null > Packages
    gzip -fk Packages
)

echo "Local Atlantis package feed written to $REPO_DIR"
PS>$global:?
True
